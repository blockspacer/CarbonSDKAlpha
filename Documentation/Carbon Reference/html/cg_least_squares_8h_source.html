58  cgPlane computePlane( ) const
59  {
60  const float denom = -2*x*xz*z + xx*z*z + x*x*zz + samples*(xz*xz - xx*zz);
61 
62  // Compute the summed plane normal
63  cgVector3 normal;
64  normal.x = xy*(-z*z + samples*zz) - samples*xz*zy + x*z*zy + (xz*z - x*zz)*y;
65  normal.y = 1.0f;
66  normal.z = -x*x*zy + samples*(-xz*xy + xx*zy) - xx*z*y + x*(xy*z + xz*y);
67 
68  // (Optimization) Compute inverse values to allow us to multiply instead of divide
69  const float denomRecip = 1.0f / denom;
70  const float samplesRecip = 1.0f / samples;
71 
72  // Compute the final plane normal
73  normal.x *= denomRecip;
74  normal.z *= denomRecip;
75  cgVector3::normalize( normal, normal );
76 
77  // Compute the average center of the data set
78  cgVector3 pos;
79  pos.x = x * samplesRecip;
80  pos.y = y * samplesRecip;
81  pos.z = z * samplesRecip;
82 
83  // Construct final plane
84  return cgPlane( normal.x, normal.y, normal.z, -cgVector3::dot( pos, normal ) );
85  }
