272 class cgSceneUpdateFIFO
273 {
274 public:
275  //-------------------------------------------------------------------------
276  // Constructors & Destructors
277  //-------------------------------------------------------------------------
278  cgSceneUpdateFIFO( size_t size )
279  {
280  mCount = 0;
281  mCurrent = 0;
282  mBottom = 0;
283  mSize = size;
284  mBuffer = new cgObjectNode*[size];
285  }
286 
287  ~cgSceneUpdateFIFO( )
288  {
289  clear();
290  }
291 
292  //-------------------------------------------------------------------------
293  // Public Methods
294  //-------------------------------------------------------------------------
295  cgObjectNode ** push( cgObjectNode * node )
296  {
297  ++mCount;
298  cgObjectNode ** result = &mBuffer[mCurrent];
299  mBuffer[mCurrent] = node;
300  ++mCurrent;
301  if ( mCurrent == mSize )
302  mCurrent = 0;
303  return result;
304  };
305 
306  cgObjectNode * pop()
307  {
308  while ( mCurrent != mBottom )
309  {
310  --mCount;
311  cgObjectNode * result = mBuffer[mBottom];
312  ++mBottom;
313  if ( mBottom == mSize )
314  mBottom = 0;
315  if ( result )
316  return result;
317  }
318  return CG_NULL;
319  }
320 
321  bool flush( cgObjectNode * node )
322  {
323  if ( mCurrent == mBottom )
324  return false;
325 
326  for ( size_t i = mBottom; i != mCurrent; )
327  {
328  if ( mBuffer[i] == node )
329  {
330  mBuffer[i] = CG_NULL;
331  return true;
332  }
333  if ( ++i == mSize )
334  i = 0;
335 
336  } // Next
337  return false;
338  }
339 
340  //-------------------------------------------------------------------------
341  // Name : clear ()
345  //-------------------------------------------------------------------------
346  void clear( )
347  {
348  delete []mBuffer;
349  mCount = 0;
350  mCurrent = 0;
351  mBottom = 0;
352  mSize = 0;
353  mBuffer = CG_NULL;
354  }
355 
356  //-------------------------------------------------------------------------
357  // Name : getEntryCount ()
361  //-------------------------------------------------------------------------
362  inline size_t getEntryCount( ) const
363  {
364  return mCount;
365  }
366 
367 private:
368  //-------------------------------------------------------------------------
369  // Private Variables
370  //-------------------------------------------------------------------------
371  size_t mCount;
372  size_t mCurrent;
373  size_t mBottom;
374  size_t mSize;
375  cgObjectNode ** mBuffer;
376 };
