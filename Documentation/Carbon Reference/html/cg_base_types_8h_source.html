148 struct CGE_API cgHalf
149 {
150 public:
151  cgHalf() {};
152  cgHalf( cgFloat f )
153  {
154  const cgUInt32 data = (cgUInt32&)f;
155  const cgUInt32 signBit = (data >> 31);
156  cgUInt32 exponent = ((data >> 23) & 0xFF);
157  cgUInt32 mantissa = (data & 0x7FFFFF);
158 
159  // Handle cases
160  if ( exponent == 255 ) // NaN or inf
161  exponent = 31;
162  else if ( exponent < 102 ) // (127-15)-10
163  exponent = mantissa = 0;
164  else if ( exponent >= 143 ) // 127+(31-15)
165  {
166  exponent = 31;
167  mantissa = 0;
168  }
169  else if( exponent <= 112 ) // 127-15
170  {
171  mantissa |= (1<<23);
172  mantissa = mantissa >> (1 + (112 - exponent));
173  exponent = 0;
174  }
175  else
176  exponent -= 112;
177 
178  // Store
179  _data = (cgUInt16)((signBit << 15) | (exponent << 10) | (mantissa >> 13));
180  }
181 
182  cgHalf( const cgHalf & h ) :
183  _data( h._data ) {}
184 
185  // Inline operators
186  inline bool operator == ( const cgHalf & h ) const
187  {
188  return _data == h._data;
189  }
190  inline bool operator != ( const cgHalf & h ) const
191  {
192  return _data != h._data;
193  }
194  inline operator cgFloat() const
195  {
196  const cgUInt32 signBit = (_data >> 15);
197  cgUInt32 exponent = ((_data >> 10) & 0x1F);
198  cgUInt32 mantissa = (_data & 0x03FF);
199 
200  // Handle cases
201  if(exponent == 31)
202  exponent = 255;
203  else if(exponent == 0)
204  exponent = 0;
205  else
206  exponent += 112;
207 
208  // Convert
209  const cgUInt32 data = (signBit << 31) | (exponent << 23) | (mantissa << 13);
210  return (cgFloat&)data;
211  }
212 
213 private:
214  // Private Members
215  cgUInt16 _data;
216 
217 }; // End Struct cgHalf
